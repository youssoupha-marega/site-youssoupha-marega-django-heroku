## 1. creer une application name_app = "blogue"
Python manage.py startapp blogue

## 2. ajouter l'application au fichier myapp/settings.py
INSTALLED_APPS = [
    ...
    'blogue',
]

## 3. Crée un fichier blog/urls.py
# blog/urls.py
from django.urls import path
from . import views

urlpatterns = [
    path('', views.blogue_acceuil, name='blogue_acceuil'),           # /blog/
    #path('article-1/', views.blog_article_1, name='blog_article_1'),
    #path('article-2/', views.blog_article_2, name='blog_article_2'),
    #path('article-3/', views.blog_article_3, name='blog_article_3'),
]

## 4. Ajouter les routes de l'application blogue dans le fichier myproject/urls.py
urlpatterns = [
    path('admin/', admin.site.urls),
    path('', include('myapp.urls')),  # Inclure les routes de myapp
    path('blogue/', include('blogue.urls')), # Inclure les routes de l'application blogue
]

## 5. Creer les templates (creation dossier templates/blogue/)
    #blog/templates/blogue/acceuil.html

    <body>

python manage.py makemigrations
